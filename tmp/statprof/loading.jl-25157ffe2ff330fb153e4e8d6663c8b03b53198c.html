6 # Cross-platform case-sensitive path canonicalization
7
8 if Sys.isunix() && !Sys.isapple()
9 # assume case-sensitive filesystems, don't have to do anything
10 isfile_casesensitive(path) = isaccessiblefile(path)
11 elseif Sys.iswindows()
12 # GetLongPathName Win32 function returns the case-preserved filename on NTFS.
13 function isfile_casesensitive(path)
14 isaccessiblefile(path) || return false # Fail fast
15 basename(Filesystem.longpath(path)) == basename(path)
16 end
17 elseif Sys.isapple()
18 # HFS+ filesystem is case-preserving. The getattrlist API returns
19 # a case-preserved filename. In the rare event that HFS+ is operating
20 # in case-sensitive mode, this will still work but will be redundant.
21
22 # Constants from <sys/attr.h>
23 const ATRATTR_BIT_MAP_COUNT = 5
24 const ATTR_CMN_NAME = 1
25 const BITMAPCOUNT = 1
26 const COMMONATTR = 5
27 const FSOPT_NOFOLLOW = 1 # Don't follow symbolic links
28
29 const attr_list = zeros(UInt8, 24)
30 attr_list[BITMAPCOUNT] = ATRATTR_BIT_MAP_COUNT
31 attr_list[COMMONATTR] = ATTR_CMN_NAME
32
33 # This essentially corresponds to the following C code:
34 # attrlist attr_list;
35 # memset(&attr_list, 0, sizeof(attr_list));
36 # attr_list.bitmapcount = ATTR_BIT_MAP_COUNT;
37 # attr_list.commonattr = ATTR_CMN_NAME;
38 # struct Buffer {
39 # u_int32_t total_length;
40 # u_int32_t filename_offset;
41 # u_int32_t filename_length;
42 # char filename[max_filename_length];
43 # };
44 # Buffer buf;
45 # getattrpath(path, &attr_list, &buf, sizeof(buf), FSOPT_NOFOLLOW);
46 function isfile_casesensitive(path)
47 isaccessiblefile(path) || return false
48 path_basename = String(basename(path))
49 local casepreserved_basename
50 header_size = 12
51 buf = Vector{UInt8}(undef, length(path_basename) + header_size + 1)
52 while true
53 ret = ccall(:getattrlist, Cint,
54 (Cstring, Ptr{Cvoid}, Ptr{Cvoid}, Csize_t, Culong),
55 path, attr_list, buf, sizeof(buf), FSOPT_NOFOLLOW)
56 systemerror(:getattrlist, ret ≠ 0)
57 filename_length = GC.@preserve buf unsafe_load(
58 convert(Ptr{UInt32}, pointer(buf) + 8))
59 if (filename_length + header_size) > length(buf)
60 resize!(buf, filename_length + header_size)
61 continue
62 end
63 casepreserved_basename =
64 view(buf, (header_size+1):(header_size+filename_length-1))
65 break
66 end
67 # Hack to compensate for inability to create a string from a subarray with no allocations.
68 codeunits(path_basename) == casepreserved_basename && return true
69
70 # If there is no match, it's possible that the file does exist but HFS+
71 # performed unicode normalization. See https://developer.apple.com/library/mac/qa/qa1235/_index.html.
72 isascii(path_basename) && return false
73 codeunits(Unicode.normalize(path_basename, :NFD)) == casepreserved_basename
74 end

233 function get_updated_dict(p::TOML.Parser, f::CachedTOMLDict)
234 s = stat(f.path)
235 time_since_cached = time() - f.mtime
236 rough_mtime_granularity = 0.1 # seconds
237 # In case the file is being updated faster than the mtime granularity,
238 # and have the same size after the update we might miss that it changed. Therefore
239 # always check the hash in case we recently created the cache.
240 if time_since_cached < rough_mtime_granularity || s.inode != f.inode || s.mtime != f.mtime || f.size != s.size
241 content = read(f.path)
242 new_hash = _crc32c(content)
243 if new_hash != f.hash
244 f.inode = s.inode
245 f.mtime = s.mtime
246 f.size = s.size
247 f.hash = new_hash
248 TOML.reinit!(p, String(content); filepath=f.path)
249 return f.d = TOML.parse(p)
250 end
251 end
252 return f.d
253 end

981 # returns `nothing` if require found a precompile cache for this sourcepath, but couldn't load it
982 # returns the set of modules restored if the cache load succeeded
983 @constprop :none function _require_search_from_serialized(pkg::PkgId, sourcepath::String, build_id::UInt64)
984 assert_havelock(require_lock)
985 paths = find_all_in_cache_path(pkg)
986 for path_to_try in paths::Vector{String}
987 staledeps = stale_cachefile(pkg, build_id, sourcepath, path_to_try)
988 if staledeps === true
989 continue
990 end
991 staledeps = staledeps::Vector{Any}
992 # finish checking staledeps module graph
993 for i in 1:length(staledeps)
994 dep = staledeps[i]
995 dep isa Module && continue
996 modpath, modkey, modbuild_id = dep::Tuple{String, PkgId, UInt64}
997 modpaths = find_all_in_cache_path(modkey)
998 modfound = false
999 for modpath_to_try in modpaths::Vector{String}
1000 modstaledeps = stale_cachefile(modkey, modbuild_id, modpath, modpath_to_try)
1001 if modstaledeps === true
1002 continue
1003 end
1004 modstaledeps = modstaledeps::Vector{Any}
1005 staledeps[i] = (modpath, modkey, modpath_to_try, modstaledeps)
1006 modfound = true
1007 break
1008 end
1009 if !modfound
1010 @debug "Rejecting cache file $path_to_try because required dependency $modkey with build ID $modbuild_id is missing from the cache."
1011 staledeps = true
1012 break
1013 end
1014 end
1015 if staledeps === true
1016 continue
1017 end
1018 try
1019 touch(path_to_try) # update timestamp of precompilation file
1020 catch ex # file might be read-only and then we fail to update timestamp, which is fine
1021 ex isa IOError || rethrow()
1022 end
1023 # finish loading module graph into staledeps
1024 for i in 1:length(staledeps)
1025 dep = staledeps[i]
1026 dep isa Module && continue
1027 modpath, modkey, modpath_to_try, modstaledeps = dep::Tuple{String, PkgId, String, Vector{Any}}
1028 dep = _tryrequire_from_serialized(modkey, modpath_to_try, modpath, modstaledeps)
1029 if !isa(dep, Module)
1030 @debug "Rejecting cache file $path_to_try because required dependency $modkey failed to load from cache file for $modpath." exception=dep
1031 staledeps = true
1032 break
1033 end
1034 staledeps[i] = dep
1035 end
1036 if staledeps === true
1037 continue
1038 end
1039 restored = _include_from_serialized(pkg, path_to_try, staledeps)
1040 if !isa(restored, Module)
1041 @debug "Deserialization checks failed while attempting to load cache from $path_to_try" exception=restored
1042 else
1043 return restored
1044 end
1045 end
1046 return
1047 end

1700 end

1977 """
1978 recursive_prefs_merge(base::Dict, overrides::Dict...)
1979
1980 Helper function to merge preference dicts recursively, honoring overrides in nested
1981 dictionaries properly.
1982 """
1983 function recursive_prefs_merge(base::Dict{String, Any}, overrides::Dict{String, Any}...)
1984 new_base = Base._typeddict(base, overrides...)
1985
1986 for override in overrides
1987 # Clear entries are keys that should be deleted from any previous setting.
1988 override_clear = get(override, "__clear__", nothing)
1989 if override_clear isa Vector{String}
1990 for k in override_clear
1991 delete!(new_base, k)
1992 end
1993 end
1994
1995 for (k, override_k) in override
1996 # Note that if `base` has a mapping that is _not_ a `Dict`, and `override`
1997 new_base_k = get(new_base, k, nothing)
1998 if new_base_k isa Dict{String, Any} && override_k isa Dict{String, Any}
1999 new_base[k] = recursive_prefs_merge(new_base_k, override_k)
2000 else
2001 new_base[k] = override_k
2002 end
2003 end
2004 end
2005 return new_base
2006 end

2065 @constprop :none function stale_cachefile(modkey::PkgId, build_id::UInt64, modpath::String, cachefile::String; ignore_loaded::Bool = false)
2066 io = open(cachefile, "r")
2067 try
2068 if !isvalid_cache_header(io)
2069 @debug "Rejecting cache file $cachefile due to it containing an invalid cache header"
2070 return true # invalid cache file
2071 end
2072 modules, (includes, requires), required_modules, srctextpos, prefs, prefs_hash = parse_cache_header(io)
2073 if isempty(modules)
2074 return true # ignore empty file
2075 end
2076 id = first(modules)
2077 if id.first != modkey && modkey != PkgId("")
2078 @debug "Rejecting cache file $cachefile for $modkey since it is for $id instead"
2079 return true
2080 end
2081 if build_id != UInt64(0) && id.second != build_id
2082 @debug "Ignoring cache file $cachefile for $modkey since it is does not provide desired build_id"
2083 return true
2084 end
2085 id = id.first
2086 modules = Dict{PkgId, UInt64}(modules)
2087
2088 # Check if transitive dependencies can be fulfilled
2089 ndeps = length(required_modules)
2090 depmods = Vector{Any}(undef, ndeps)
2091 for i in 1:ndeps
2092 req_key, req_build_id = required_modules[i]
2093 # Module is already loaded
2094 if root_module_exists(req_key)
2095 M = root_module(req_key)
2096 if PkgId(M) == req_key && module_build_id(M) === req_build_id
2097 depmods[i] = M
2098 elseif ignore_loaded
2099 # Used by Pkg.precompile given that there it's ok to precompile different versions of loaded packages
2100 @goto locate_branch
2101 else
2102 @debug "Rejecting cache file $cachefile because module $req_key is already loaded and incompatible."
2103 return true # Won't be able to fulfill dependency
2104 end
2105 else
2106 @label locate_branch
2107 path = locate_package(req_key)
2108 if path === nothing
2109 @debug "Rejecting cache file $cachefile because dependency $req_key not found."
2110 return true # Won't be able to fulfill dependency
2111 end
2112 depmods[i] = (path, req_key, req_build_id)
2113 end
2114 end
2115
2116 # check if this file is going to provide one of our concrete dependencies
2117 # or if it provides a version that conflicts with our concrete dependencies
2118 # or neither
2119 skip_timecheck = false
2120 for (req_key, req_build_id) in _concrete_dependencies
2121 build_id = get(modules, req_key, UInt64(0))
2122 if build_id !== UInt64(0)
2123 if build_id === req_build_id
2124 skip_timecheck = true
2125 break
2126 end
2127 @debug "Rejecting cache file $cachefile because it provides the wrong build_id (got $build_id) for $req_key (want $req_build_id)"
2128 return true # cachefile doesn't provide the required version of the dependency
2129 end
2130 end
2131
2132 # now check if this file is fresh relative to its source files
2133 if !skip_timecheck
2134 if !samefile(includes[1].filename, modpath)
2135 @debug "Rejecting cache file $cachefile because it is for file $(includes[1].filename) not file $modpath"
2136 return true # cache file was compiled from a different path
2137 end
2138 for (modkey, req_modkey) in requires
2139 # verify that `require(modkey, name(req_modkey))` ==> `req_modkey`
2140 if identify_package(modkey, req_modkey.name) != req_modkey
2141 @debug "Rejecting cache file $cachefile because uuid mapping for $modkey => $req_modkey has changed"
2142 return true
2143 end
2144 end
2145 for chi in includes
2146 f, ftime_req = chi.filename, chi.mtime
2147 ftime = mtime(f)
2148 is_stale = ( ftime != ftime_req ) &&
2149 ( ftime != floor(ftime_req) ) && # Issue #13606, PR #13613: compensate for Docker images rounding mtimes
2150 ( ftime != ceil(ftime_req) ) && # PR: #47433 Compensate for CirceCI's truncating of timestamps in its caching
2151 ( ftime != trunc(ftime_req, digits=6) ) && # Issue #20837, PR #20840: compensate for GlusterFS truncating mtimes to microseconds
2152 ( ftime != 1.0 ) && # PR #43090: provide compatibility with Nix mtime.
2153 !( 0 < (ftime_req - ftime) < 1e-6 ) # PR #45552: Compensate for Windows tar giving mtimes that may be incorrect by up to one microsecond
2154 if is_stale
2155 @debug "Rejecting stale cache file $cachefile (mtime $ftime_req) because file $f (mtime $ftime) has changed"
2156 return true
2157 end
2158 end
2159 end
2160
2161 if !isvalid_file_crc(io)
2162 @debug "Rejecting cache file $cachefile because it has an invalid checksum"
2163 return true
2164 end
2165
2166 curr_prefs_hash = get_preferences_hash(id.uuid, prefs)
2167 if prefs_hash != curr_prefs_hash
2168 @debug "Rejecting cache file $cachefile because preferences hash does not match 0x$(string(prefs_hash, base=16)) != 0x$(string(curr_prefs_hash, base=16))"
2169 return true
2170 end
2171
2172 return depmods # fresh cachefile
2173 finally
2174 close(io)
2175 end
2176 end
