295 function eval_test(evaluated::Expr, quoted::Expr, source::LineNumberNode, negate::Bool=false)
296 evaled_args = evaluated.args
297 quoted_args = quoted.args
298 n = length(evaled_args)
299 kw_suffix = ""
300 if evaluated.head === :comparison
301 args = evaled_args
302 res = true
303 i = 1
304 while i < n
305 a, op, b = args[i], args[i+1], args[i+2]
306 if res
307 res = op(a, b)
308 end
309 quoted_args[i] = a
310 quoted_args[i+2] = b
311 i += 2
312 end
313
314 elseif evaluated.head === :call
315 op = evaled_args[1]
316 kwargs = (evaled_args[2]::Expr).args # Keyword arguments from `Expr(:parameters, ...)`
317 args = evaled_args[3:n]
318
319 res = op(args...; kwargs...)
320
321 # Create "Evaluated" expression which looks like the original call but has all of
322 # the arguments evaluated
323 func_sym = quoted_args[1]::Union{Symbol,Expr}
324 if isempty(kwargs)
325 quoted = Expr(:call, func_sym, args...)
326 elseif func_sym === :≈ && !res
327 quoted = Expr(:call, func_sym, args...)
328 kw_suffix = " ($(join(["$k=$v" for (k, v) in kwargs], ", ")))"
329 else
330 kwargs_expr = Expr(:parameters, [Expr(:kw, k, v) for (k, v) in kwargs]...)
331 quoted = Expr(:call, func_sym, kwargs_expr, args...)
332 end
333 else
334 throw(ArgumentError("Unhandled expression type: $(evaluated.head)"))
335 end
336
337 if negate
338 res = !res
339 quoted = Expr(:call, :!, quoted)
340 end
341
342 Returned(res,
343 # stringify arguments in case of failure, for easy remote printing
344 res === true ? quoted : sprint(print, quoted, context=(:limit => true)) * kw_suffix,
345 source)
346 end

371 """
372 @test ex
373 @test f(args...) key=val ...
374 @test ex broken=true
375 @test ex skip=true
376
377 Test that the expression `ex` evaluates to `true`.
378 If executed inside a `@testset`, return a `Pass` `Result` if it does, a `Fail` `Result` if it is
379 `false`, and an `Error` `Result` if it could not be evaluated.
380 If executed outside a `@testset`, throw an exception instead of returning `Fail` or `Error`.
381
382 # Examples
383 ```jldoctest
384 julia> @test true
385 Test Passed
386
387 julia> @test [1, 2] + [2, 1] == [3, 3]
388 Test Passed
389 ```
390
391 The `@test f(args...) key=val...` form is equivalent to writing
392 `@test f(args..., key=val...)` which can be useful when the expression
393 is a call using infix syntax such as approximate comparisons:
394
395 ```jldoctest
396 julia> @test π ≈ 3.14 atol=0.01
397 Test Passed
398 ```
399
400 This is equivalent to the uglier test `@test ≈(π, 3.14, atol=0.01)`.
401 It is an error to supply more than one expression unless the first
402 is a call expression and the rest are assignments (`k=v`).
403
404 You can use any key for the `key=val` arguments, except for `broken` and `skip`,
405 which have special meanings in the context of `@test`:
406
407 * `broken=cond` indicates a test that should pass but currently consistently
408 fails when `cond==true`. Tests that the expression `ex` evaluates to `false`
409 or causes an exception. Returns a `Broken` `Result` if it does, or an `Error`
410 `Result` if the expression evaluates to `true`. Regular `@test ex` is
411 evaluated when `cond==false`.
412 * `skip=cond` marks a test that should not be executed but should be included in
413 test summary reporting as `Broken`, when `cond==true`. This can be useful for
414 tests that intermittently fail, or tests of not-yet-implemented functionality.
415 Regular `@test ex` is evaluated when `cond==false`.
416
417 # Examples
418
419 ```jldoctest
420 julia> @test 2 + 2 ≈ 6 atol=1 broken=true
421 Test Broken
422 Expression: ≈(2 + 2, 6, atol = 1)
423
424 julia> @test 2 + 2 ≈ 5 atol=1 broken=false
425 Test Passed
426
427 julia> @test 2 + 2 == 5 skip=true
428 Test Broken
429 Skipped: 2 + 2 == 5
430
431 julia> @test 2 + 2 == 4 skip=false
432 Test Passed
433 ```

436 The `broken` and `skip` keyword arguments require at least Julia 1.7.

440 broken = [kw.args[2] for kw in kws if kw.args[1] === :broken]

531 # macro to get results of the test expression.
532 # In the special case of a comparison, e.g. x == 5, generate code to
533 # evaluate each term in the comparison individually so the results
534 # can be displayed nicely.
535 function get_test_result(ex, source)
536 negate = QuoteNode(false)
537 orig_ex = ex
538 # Evaluate `not` wrapped functions separately for pretty-printing failures
539 if isa(ex, Expr) && ex.head === :call && length(ex.args) == 2 && ex.args[1] === :!
540 negate = QuoteNode(true)
541 ex = ex.args[2]
542 end
543 # Normalize non-dot comparison operator calls to :comparison expressions
544 is_splat = x -> isa(x, Expr) && x.head === :...
545 if isa(ex, Expr) && ex.head === :call && length(ex.args) == 3 &&
546 first(string(ex.args[1])) != '.' && !is_splat(ex.args[2]) && !is_splat(ex.args[3]) &&
547 (ex.args[1] === :(==) || Base.operator_precedence(ex.args[1]) == comparison_prec)
548 ex = Expr(:comparison, ex.args[2], ex.args[1], ex.args[3])
549
550 # Mark <: and >: as :comparison expressions
551 elseif isa(ex, Expr) && length(ex.args) == 2 &&
552 !is_splat(ex.args[1]) && !is_splat(ex.args[2]) &&
553 Base.operator_precedence(ex.head) == comparison_prec
554 ex = Expr(:comparison, ex.args[1], ex.head, ex.args[2])
555 end
556 if isa(ex, Expr) && ex.head === :comparison
557 # pass all terms of the comparison to `eval_comparison`, as an Expr
558 escaped_terms = [esc(arg) for arg in ex.args]
559 quoted_terms = [QuoteNode(arg) for arg in ex.args]
560 testret = :(eval_test(
561 Expr(:comparison, $(escaped_terms...)),
562 Expr(:comparison, $(quoted_terms...)),
563 $(QuoteNode(source)),
564 $negate,
565 ))
566 elseif isa(ex, Expr) && ex.head === :call && ex.args[1] in DISPLAY_FAILED
567 escaped_func = esc(ex.args[1])
568 quoted_func = QuoteNode(ex.args[1])
569
570 escaped_args = []
571 escaped_kwargs = []
572
573 # Keywords that occur before `;`. Note that the keywords are being revised into
574 # a form we can splat.
575 for a in ex.args[2:end]
576 if isa(a, Expr) && a.head === :kw
577 push!(escaped_kwargs, Expr(:call, :(=>), QuoteNode(a.args[1]), esc(a.args[2])))
578 end
579 end
580
581 # Keywords that occur after ';'
582 parameters_expr = ex.args[2]
583 if isa(parameters_expr, Expr) && parameters_expr.head === :parameters
584 for a in parameters_expr.args
585 if isa(a, Expr) && a.head === :kw
586 push!(escaped_kwargs, Expr(:call, :(=>), QuoteNode(a.args[1]), esc(a.args[2])))
587 elseif isa(a, Expr) && a.head === :...
588 push!(escaped_kwargs, Expr(:..., esc(a.args[1])))
589 elseif isa(a, Expr) && a.head === :.
590 push!(escaped_kwargs, Expr(:call, :(=>), QuoteNode(a.args[2].value), esc(Expr(:., a.args[1], QuoteNode(a.args[2].value)))))
591 elseif isa(a, Symbol)
592 push!(escaped_kwargs, Expr(:call, :(=>), QuoteNode(a), esc(a)))
593 end
594 end
595 end
596
597 # Positional arguments
598 for a in ex.args[2:end]
599 isa(a, Expr) && a.head in (:kw, :parameters) && continue
600
601 if isa(a, Expr) && a.head === :...
602 push!(escaped_args, Expr(:..., esc(a.args[1])))
603 else
604 push!(escaped_args, esc(a))
605 end
606 end

609 Expr(:call, $escaped_func, Expr(:parameters, $(escaped_kwargs...)), $(escaped_args...)),
610 Expr(:call, $quoted_func),
611 $(QuoteNode(source)),
612 $negate,
613 ))
614 else
615 testret = :(Returned($(esc(orig_ex)), nothing, $(QuoteNode(source))))
616 end
617 result = quote
618 try

621 _e isa InterruptException && rethrow()

629 # An internal function, called by the code generated by the @test
630 # macro to actually perform the evaluation and manage the result.
631 function do_test(result::ExecutionResult, orig_expr)
632 # get_testset() returns the most recently added test set
633 # We then call record() with this test set and the test result
634 if isa(result, Returned)
635 # expr, in the case of a comparison, will contain the
636 # comparison with evaluated values of each term spliced in.
637 # For anything else, just contains the test expression.
638 # value is the evaluated value of the whole test expression.
639 # Ideally it is true, but it may be false or non-Boolean.
640 value = result.value
641 testres = if isa(value, Bool)
642 # a true value Passes
643 value ? Pass(:test, orig_expr, result.data, value, result.source) :
644 Fail(:test, orig_expr, result.data, value, result.source)
645 else
646 # If the result is non-Boolean, this counts as an Error
647 Error(:test_nonbool, orig_expr, value, nothing, result.source)
648 end
649 else
650 # The predicate couldn't be evaluated without throwing an
651 # exception, so that is an Error and not a Fail
652 @assert isa(result, Threw)
653 testres = Error(:test_error, orig_expr, result.exception, result.backtrace::Vector{Any}, result.source)
654 end
655 isa(testres, Pass) || ccall(:jl_breakpoint, Cvoid, (Any,), result)
656 record(get_testset(), testres)
657 end

660 testres = Broken(:test, orig_expr)
661 # Assume the test is broken and only change if the result is true
662 if isa(result, Returned)
663 value = result.value
664 if isa(value, Bool) && value
665 testres = Error(:test_unbroken, orig_expr, value, nothing, result.source)
666 end
667 end

679 a matching function,
680 or a value (which will be tested for equality by comparing fields).
681 Note that `@test_throws` does not support a trailing keyword form.
682
683 !!! compat "Julia 1.8"
684 The ability to specify anything other than a type or a value as `exception` requires Julia v1.8 or later.
685
686 # Examples
687 ```jldoctest
688 julia> @test_throws BoundsError [1, 2, 3][4]
689 Test Passed
690 Thrown: BoundsError
691
692 julia> @test_throws DimensionMismatch [1, 2, 3] + [1, 2]
693 Test Passed
694 Thrown: DimensionMismatch
695
696 julia> @test_throws "Try sqrt(Complex" sqrt(-1)
697 Test Passed
698 Message: "DomainError with -1.0:\\nsqrt will only return a complex result if called with a complex argument. Try sqrt(Complex(x))."
699 ```
700
701 In the final example, instead of matching a single string it could alternatively have been performed with:
702
703 - `["Try", "Complex"]` (a list of strings)
704 - `r"Try sqrt\\([Cc]omplex"` (a regular expression)
705 - `str -> occursin("complex", str)` (a matching function)
706 """
707 macro test_throws(extype, ex)
708 orig_ex = Expr(:inert, ex)
709 result = quote
710 try
711 Returned($(esc(ex)), nothing, $(QuoteNode(__source__)))
712 catch _e
713 if $(esc(extype)) != InterruptException && _e isa InterruptException
714 rethrow()
715 end
716 Threw(_e, nothing, $(QuoteNode(__source__)))
717 end
718 end
719 Base.remove_linenums!(result)
720 :(do_test_throws($result, $orig_ex, $(esc(extype))))
721 end

723 const MACROEXPAND_LIKE = Symbol.(("@macroexpand", "@macroexpand1", "macroexpand"))
724
725 # An internal function, called by the code generated by @test_throws
726 # to evaluate and catch the thrown exception - if it exists
727 function do_test_throws(result::ExecutionResult, orig_expr, extype)
728 if isa(result, Threw)
729 # Check that the right type of exception was thrown
730 success = false
731 message_only = false
732 exc = result.exception
733 # NB: Throwing LoadError from macroexpands is deprecated, but in order to limit
734 # the breakage in package tests we add extra logic here.
735 from_macroexpand =
736 orig_expr isa Expr &&
737 orig_expr.head in (:call, :macrocall) &&
738 orig_expr.args[1] in MACROEXPAND_LIKE
739 if isa(extype, Type)
740 success =
741 if from_macroexpand && extype == LoadError && exc isa Exception
742 Base.depwarn("macroexpand no longer throws a LoadError so `@test_throws LoadError ...` is deprecated and passed without checking the error type!", :do_test_throws)
743 true
744 else
745 isa(exc, extype)
746 end
747 elseif isa(extype, Exception) || !isa(exc, Exception)
748 if extype isa LoadError && !(exc isa LoadError) && typeof(extype.error) == typeof(exc)
749 extype = extype.error # deprecated
750 end
751 if isa(exc, typeof(extype))
752 success = true
753 for fld in 1:nfields(extype)
754 if !isequal(getfield(extype, fld), getfield(exc, fld))
755 success = false
756 break
757 end
758 end
759 end
760 else
761 message_only = true
762 exc = sprint(showerror, exc)
763 success = contains_warn(exc, extype)
764 exc = repr(exc)
765 if isa(extype, AbstractString)
766 extype = repr(extype)
767 elseif isa(extype, Function)
768 extype = "< match function >"
769 end
770 end
771 if success
772 testres = Pass(:test_throws, orig_expr, extype, exc, result.source, message_only)
773 else
774 testres = Fail(:test_throws_wrong, orig_expr, extype, exc, result.source, message_only)
775 end
776 else
777 testres = Fail(:test_throws_nothing, orig_expr, extype, nothing, result.source)
778 end
779 record(get_testset(), testres)
780 end

1014 function print_test_results(ts::DefaultTestSet, depth_pad=0)
1015 # Calculate the overall number for each type so each of
1016 # the test result types are aligned
1017 passes, fails, errors, broken, c_passes, c_fails, c_errors, c_broken, duration = get_test_counts(ts)
1018 total_pass = passes + c_passes
1019 total_fail = fails + c_fails
1020 total_error = errors + c_errors
1021 total_broken = broken + c_broken
1022 dig_pass = total_pass > 0 ? ndigits(total_pass) : 0
1023 dig_fail = total_fail > 0 ? ndigits(total_fail) : 0
1024 dig_error = total_error > 0 ? ndigits(total_error) : 0
1025 dig_broken = total_broken > 0 ? ndigits(total_broken) : 0
1026 total = total_pass + total_fail + total_error + total_broken
1027 dig_total = total > 0 ? ndigits(total) : 0
1028 # For each category, take max of digits and header width if there are
1029 # tests of that type
1030 pass_width = dig_pass > 0 ? max(length("Pass"), dig_pass) : 0
1031 fail_width = dig_fail > 0 ? max(length("Fail"), dig_fail) : 0
1032 error_width = dig_error > 0 ? max(length("Error"), dig_error) : 0
1033 broken_width = dig_broken > 0 ? max(length("Broken"), dig_broken) : 0
1034 total_width = dig_total > 0 ? max(length("Total"), dig_total) : 0
1035 duration_width = max(length("Time"), length(duration))
1036 # Calculate the alignment of the test result counts by
1037 # recursively walking the tree of test sets
1038 align = max(get_alignment(ts, 0), length("Test Summary:"))
1039 # Print the outer test set header once
1040 pad = total == 0 ? "" : " "
1041 printstyled(rpad("Test Summary:", align, " "), " |", pad; bold=true)
1042 if pass_width > 0
1043 printstyled(lpad("Pass", pass_width, " "), " "; bold=true, color=:green)
1044 end
1045 if fail_width > 0
1046 printstyled(lpad("Fail", fail_width, " "), " "; bold=true, color=Base.error_color())
1047 end
1048 if error_width > 0
1049 printstyled(lpad("Error", error_width, " "), " "; bold=true, color=Base.error_color())
1050 end
1051 if broken_width > 0
1052 printstyled(lpad("Broken", broken_width, " "), " "; bold=true, color=Base.warn_color())
1053 end
1054 if total_width > 0
1055 printstyled(lpad("Total", total_width, " "), " "; bold=true, color=Base.info_color())
1056 end
1057 if ts.showtiming
1058 printstyled(lpad("Time", duration_width, " "); bold=true)
1059 end
1060 println()
1061 # Recursively print a summary at every level
1062 print_counts(ts, depth_pad, align, pass_width, fail_width, error_width, broken_width, total_width, duration_width, ts.showtiming)
1063 end

1068 # Called at the end of a @testset, behaviour depends on whether
1069 # this is a child of another testset, or the "root" testset
1070 function finish(ts::DefaultTestSet)
1071 ts.time_end = time()
1072 # If we are a nested test set, do not print a full summary
1073 # now - let the parent test set do the printing
1074 if get_testset_depth() != 0
1075 # Attach this test set to the parent test set
1076 parent_ts = get_testset()
1077 record(parent_ts, ts)
1078 return ts
1079 end
1080 passes, fails, errors, broken, c_passes, c_fails, c_errors, c_broken, duration = get_test_counts(ts)
1081 total_pass = passes + c_passes
1082 total_fail = fails + c_fails
1083 total_error = errors + c_errors
1084 total_broken = broken + c_broken
1085 total = total_pass + total_fail + total_error + total_broken
1086
1087 if TESTSET_PRINT_ENABLE[]
1088 print_test_results(ts)
1089 end
1090
1091 # Finally throw an error as we are the outermost test set
1092 if total != total_pass + total_broken
1093 # Get all the error/failures and bring them along for the ride
1094 efs = filter_errors(ts)
1095 throw(TestSetException(total_pass, total_fail, total_error, total_broken, efs))
1096 end
1097
1098 # return the testset so it is returned from the @testset macro
1099 ts
1100 end

1102 # Recursive function that finds the column that the result counts
1103 # can begin at by taking into account the width of the descriptions
1104 # and the amount of indentation. If a test set had no failures, and
1105 # no failures in child test sets, there is no need to include those
1106 # in calculating the alignment
1107 function get_alignment(ts::DefaultTestSet, depth::Int)
1108 # The minimum width at this depth is
1109 ts_width = 2*depth + length(ts.description)
1110 # If not verbose and all passing, no need to look at children
1111 !ts.verbose && !ts.anynonpass && return ts_width
1112 # Return the maximum of this width and the minimum width
1113 # for all children (if they exist)
1114 isempty(ts.results) && return ts_width
1115 child_widths = map(t->get_alignment(t, depth+1), ts.results)
1116 return max(ts_width, maximum(child_widths))
1117 end

1125 if isa(t, DefaultTestSet)

1134 # Recursive function that counts the number of test results of each
1135 # type directly in the testset, and totals across the child testsets
1136 function get_test_counts(ts::DefaultTestSet)
1137 passes, fails, errors, broken = ts.n_passed, 0, 0, 0
1138 c_passes, c_fails, c_errors, c_broken = 0, 0, 0, 0
1139 for t in ts.results
1140 isa(t, Fail) && (fails += 1)
1141 isa(t, Error) && (errors += 1)
1142 isa(t, Broken) && (broken += 1)
1143 if isa(t, DefaultTestSet)
1144 np, nf, ne, nb, ncp, ncf, nce , ncb, duration = get_test_counts(t)
1145 c_passes += np + ncp
1146 c_fails += nf + ncf
1147 c_errors += ne + nce
1148 c_broken += nb + ncb
1149 end
1150 end
1151 ts.anynonpass = (fails + errors + c_fails + c_errors > 0)
1152 duration = if isnothing(ts.time_end)
1153 ""
1154 else
1155 dur_s = ts.time_end - ts.time_start
1156 if dur_s < 60
1157 string(round(dur_s, digits = 1), "s")
1158 else
1159 m, s = divrem(dur_s, 60)
1160 s = lpad(string(round(s, digits = 1)), 4, "0")
1161 string(round(Int, m), "m", s, "s")
1162 end
1163 end
1164 return passes, fails, errors, broken, c_passes, c_fails, c_errors, c_broken, duration
1165 end

1167 # Recursive function that prints out the results at each level of
1168 # the tree of test sets
1169 function print_counts(ts::DefaultTestSet, depth, align,
1170 pass_width, fail_width, error_width, broken_width, total_width, duration_width, showtiming)
1171 # Count results by each type at this level, and recursively
1172 # through any child test sets
1173 passes, fails, errors, broken, c_passes, c_fails, c_errors, c_broken, duration = get_test_counts(ts)
1174 subtotal = passes + fails + errors + broken + c_passes + c_fails + c_errors + c_broken
1175 # Print test set header, with an alignment that ensures all
1176 # the test results appear above each other
1177 print(rpad(string(" "^depth, ts.description), align, " "), " | ")
1178
1179 np = passes + c_passes
1180 if np > 0
1181 printstyled(lpad(string(np), pass_width, " "), " ", color=:green)
1182 elseif pass_width > 0
1183 # No passes at this level, but some at another level
1184 print(lpad(" ", pass_width), " ")
1185 end
1186
1187 nf = fails + c_fails
1188 if nf > 0
1189 printstyled(lpad(string(nf), fail_width, " "), " ", color=Base.error_color())
1190 elseif fail_width > 0
1191 # No fails at this level, but some at another level
1192 print(lpad(" ", fail_width), " ")
1193 end
1194
1195 ne = errors + c_errors
1196 if ne > 0
1197 printstyled(lpad(string(ne), error_width, " "), " ", color=Base.error_color())
1198 elseif error_width > 0
1199 # No errors at this level, but some at another level
1200 print(lpad(" ", error_width), " ")
1201 end
1202
1203 nb = broken + c_broken
1204 if nb > 0
1205 printstyled(lpad(string(nb), broken_width, " "), " ", color=Base.warn_color())
1206 elseif broken_width > 0
1207 # None broken at this level, but some at another level
1208 print(lpad(" ", broken_width), " ")
1209 end
1210
1211 if np == 0 && nf == 0 && ne == 0 && nb == 0
1212 printstyled(lpad("None", total_width, " "), " ", color=Base.info_color())
1213 else
1214 printstyled(lpad(string(subtotal), total_width, " "), " ", color=Base.info_color())
1215 end
1216
1217 if showtiming
1218 printstyled(lpad(string(duration), duration_width, " "))
1219 end
1220 println()
1221
1222 # Only print results at lower levels if we had failures or if the user
1223 # wants.
1224 if (np + nb != subtotal) || (ts.verbose)
1225 for t in ts.results
1226 if isa(t, DefaultTestSet)
1227 print_counts(t, depth + 1, align,
1228 pass_width, fail_width, error_width, broken_width, total_width, duration_width, ts.showtiming)
1229 end
1230 end
1231 end

1236 function _check_testset(testsettype, testsetname)
1237 if !(testsettype isa Type && testsettype <: AbstractTestSet)
1238 error("Expected `$testsetname` to be an AbstractTestSet, it is a ",
1239 typeof(testsettype), ". ",
1240 typeof(testsettype) == String ?
1241 """
1242 To use `$testsetname` as a testset name, interpolate it into a string, e.g:
1243 @testset "\$$testsetname" begin
1244 ...
1245 end"""
1246 :
1247 ""
1248 )
1249 end
1250 end
1251
1252 """
1253 @testset [CustomTestSet] [option=val ...] ["description"] begin ... end
1254 @testset [CustomTestSet] [option=val ...] ["description \$v"] for v in (...) ... end
1255 @testset [CustomTestSet] [option=val ...] ["description \$v, \$w"] for v in (...), w in (...) ... end
1256 @testset [CustomTestSet] [option=val ...] ["description \$v, \$w"] foo()
1257
1258 Starts a new test set, or multiple test sets if a `for` loop is provided.
1259
1260 If no custom testset type is given it defaults to creating a `DefaultTestSet`.
1261 `DefaultTestSet` records all the results and, if there are any `Fail`s or
1262 `Error`s, throws an exception at the end of the top-level (non-nested) test set,
1263 along with a summary of the test results.
1264
1265 Any custom testset type (subtype of `AbstractTestSet`) can be given and it will
1266 also be used for any nested `@testset` invocations. The given options are only
1267 applied to the test set where they are given. The default test set type
1268 accepts two boolean options:
1269 - `verbose`: if `true`, the result summary of the nested testsets is shown even
1270 when they all pass (the default is `false`).
1271 - `showtiming`: if `true`, the duration of each displayed testset is shown
1272 (the default is `true`).
1273
1274 !!! compat "Julia 1.8"
1275 `@testset foo()` requires at least Julia 1.8.
1276
1277 The description string accepts interpolation from the loop indices.
1278 If no description is provided, one is constructed based on the variables.
1279 If a function call is provided, its name will be used. Explicit description strings override this behavior.
1280
1281 By default the `@testset` macro will return the testset object itself, though
1282 this behavior can be customized in other testset types. If a `for` loop is used
1283 then the macro collects and returns a list of the return values of the `finish`
1284 method, which by default will return a list of the testset objects used in
1285 each iteration.
1286
1287 Before the execution of the body of a `@testset`, there is an implicit
1288 call to `Random.seed!(seed)` where `seed` is the current seed of the global RNG.
1289 Moreover, after the execution of the body, the state of the global RNG is
1290 restored to what it was before the `@testset`. This is meant to ease
1291 reproducibility in case of failure, and to allow seamless
1292 re-arrangements of `@testset`s regardless of their side-effect on the
1293 global RNG state.
1294
1295 # Examples
1296 ```jldoctest; filter = r"trigonometric identities | 4 4 [0-9\\.]+s"
1297 julia> @testset "trigonometric identities" begin
1298 θ = 2/3*π
1299 @test sin(-θ) ≈ -sin(θ)
1300 @test cos(-θ) ≈ cos(θ)
1301 @test sin(2θ) ≈ 2*sin(θ)*cos(θ)
1302 @test cos(2θ) ≈ cos(θ)^2 - sin(θ)^2
1303 end;
1304 Test Summary: | Pass Total Time
1305 trigonometric identities | 4 4 0.2s
1306 ```
1307 """
1308 macro testset(args...)
1309 isempty(args) && error("No arguments to @testset")
1310
1311 tests = args[end]
1312
1313 # Determine if a single block or for-loop style
1314 if !isa(tests,Expr) || (tests.head !== :for && tests.head !== :block && tests.head != :call)
1315
1316 error("Expected function call, begin/end block or for loop as argument to @testset")
1317 end
1318
1319 if tests.head === :for
1320 return testset_forloop(args, tests, __source__)
1321 else
1322 return testset_beginend_call(args, tests, __source__)
1323 end
1324 end

1326 """
1327 Generate the code for a `@testset` with a function call or `begin`/`end` argument
1328 """
1329 function testset_beginend_call(args, tests, source)
1330 desc, testsettype, options = parse_testset_args(args[1:end-1])
1331 if desc === nothing
1332 if tests.head === :call
1333 desc = string(tests.args[1]) # use the function name as test name
1334 else
1335 desc = "test set"
1336 end
1337 end
1338 # If we're at the top level we'll default to DefaultTestSet. Otherwise
1339 # default to the type of the parent testset
1340 if testsettype === nothing
1341 testsettype = :(get_testset_depth() == 0 ? DefaultTestSet : typeof(get_testset()))
1342 end
1343
1344 # Generate a block of code that initializes a new testset, adds
1345 # it to the task local storage, evaluates the test(s), before
1346 # finally removing the testset and giving it a chance to take
1347 # action (such as reporting the results)
1348 ex = quote
1349 _check_testset($testsettype, $(QuoteNode(testsettype.args[1])))
1350 local ret
1351 local ts = $(testsettype)($desc; $options...)
1352 push_testset(ts)
1353 # we reproduce the logic of guardseed, but this function
1354 # cannot be used as it changes slightly the semantic of @testset,
1355 # by wrapping the body in a function
1356 local RNG = default_rng()
1357 local oldrng = copy(RNG)
1358 local oldseed = Random.GLOBAL_SEED
1359 try
1360 # RNG is re-seeded with its own seed to ease reproduce a failed test
1361 Random.seed!(Random.GLOBAL_SEED)
1362 let
1363 $(esc(tests))
1364 end
1365 catch err
1366 err isa InterruptException && rethrow()
1367 # something in the test block threw an error. Count that as an
1368 # error in this test set
1369 record(ts, Error(:nontest_error, Expr(:tuple), err, Base.current_exceptions(), $(QuoteNode(source))))
1370 finally
1371 copy!(RNG, oldrng)
1372 Random.set_global_seed!(oldseed)
1373 pop_testset()
1374 ret = finish(ts)
1375 end
1376 ret
1377 end
1378 # preserve outer location if possible
1379 if tests isa Expr && tests.head === :block && !isempty(tests.args) && tests.args[1] isa LineNumberNode
1380 ex = Expr(:block, tests.args[1], ex)
1381 end
1382 return ex
1383 end

1386 """
1387 Generate the code for a `@testset` with a `for` loop argument
1388 """
1389 function testset_forloop(args, testloop, source)
1390 # Pull out the loop variables. We might need them for generating the
1391 # description and we'll definitely need them for generating the
1392 # comprehension expression at the end
1393 loopvars = Expr[]
1394 if testloop.args[1].head === :(=)
1395 push!(loopvars, testloop.args[1])
1396 elseif testloop.args[1].head === :block
1397 for loopvar in testloop.args[1].args
1398 push!(loopvars, loopvar)
1399 end
1400 else
1401 error("Unexpected argument to @testset")
1402 end
1403
1404 desc, testsettype, options = parse_testset_args(args[1:end-1])
1405
1406 if desc === nothing
1407 # No description provided. Generate from the loop variable names
1408 v = loopvars[1].args[1]
1409 desc = Expr(:string, "$v = ", esc(v)) # first variable
1410 for l = loopvars[2:end]
1411 v = l.args[1]
1412 push!(desc.args, ", $v = ")
1413 push!(desc.args, esc(v))
1414 end
1415 end
1416
1417 if testsettype === nothing
1418 testsettype = :(get_testset_depth() == 0 ? DefaultTestSet : typeof(get_testset()))
1419 end
1420
1421 # Uses a similar block as for `@testset`, except that it is
1422 # wrapped in the outer loop provided by the user
1423 tests = testloop.args[2]
1424 blk = quote
1425 _check_testset($testsettype, $(QuoteNode(testsettype.args[1])))
1426 # Trick to handle `break` and `continue` in the test code before
1427 # they can be handled properly by `finally` lowering.
1428 if !first_iteration
1429 pop_testset()
1430 push!(arr, finish(ts))
1431 # it's 1000 times faster to copy from tmprng rather than calling Random.seed!
1432 copy!(RNG, tmprng)
1433
1434 end
1435 ts = $(testsettype)($desc; $options...)
1436 push_testset(ts)
1437 first_iteration = false
1438 try
1439 $(esc(tests))
1440 catch err
1441 err isa InterruptException && rethrow()
1442 # Something in the test block threw an error. Count that as an
1443 # error in this test set
1444 record(ts, Error(:nontest_error, Expr(:tuple), err, Base.current_exceptions(), $(QuoteNode(source))))
1445 end
1446 end
1447 quote
1448 local arr = Vector{Any}()
1449 local first_iteration = true
1450 local ts
1451 local RNG = default_rng()
1452 local oldrng = copy(RNG)
1453 local oldseed = Random.GLOBAL_SEED
1454 Random.seed!(Random.GLOBAL_SEED)
1455 local tmprng = copy(RNG)
1456 try
1457 let
1458 $(Expr(:for, Expr(:block, [esc(v) for v in loopvars]...), blk))
1459 end
1460 finally
1461 # Handle `return` in test body
1462 if !first_iteration
1463 pop_testset()
1464 push!(arr, finish(ts))
1465 end
1466 copy!(RNG, oldrng)
1467 Random.set_global_seed!(oldseed)
1468 end
1469 arr
1470 end
1471 end

1473 """
1474 Parse the arguments to the `@testset` macro to pull out the description,
1475 Testset Type, and options. Generally this should be called with all the macro
1476 arguments except the last one, which is the test expression itself.
1477 """
1478 function parse_testset_args(args)
1479 desc = nothing
1480 testsettype = nothing
1481 options = :(Dict{Symbol, Any}())
1482 for arg in args
1483 # a standalone symbol is assumed to be the test set we should use
1484 if isa(arg, Symbol)
1485 testsettype = esc(arg)
1486 # a string is the description
1487 elseif isa(arg, AbstractString) || (isa(arg, Expr) && arg.head === :string)
1488 desc = esc(arg)
1489 # an assignment is an option
1490 elseif isa(arg, Expr) && arg.head === :(=)
1491 # we're building up a Dict literal here
1492 key = Expr(:quote, arg.args[1])
1493 push!(options.args, Expr(:call, :(=>), key, esc(arg.args[2])))
1494 else
1495 error("Unexpected argument $arg to @testset")
1496 end
1497 end
1498
1499 (desc, testsettype, options)
1500 end

1553 """
1554 @inferred [AllowedType] f(x)
1555
1556 Tests that the call expression `f(x)` returns a value of the same type inferred by the
1557 compiler. It is useful to check for type stability.
1558

1560 `Error` `Result` if it finds different types.
1561
1562 Optionally, `AllowedType` relaxes the test, by making it pass when either the type of `f(x)`
1563 matches the inferred type modulo `AllowedType`, or when the return type is a subtype of
1564 `AllowedType`. This is useful when testing type stability of functions returning a small
1565 union such as `Union{Nothing, T}` or `Union{Missing, T}`.
1566
1567 ```jldoctest; setup = :(using InteractiveUtils), filter = r"begin\\n(.|\\n)*end"
1568 julia> f(a) = a > 1 ? 1 : 1.0
1569 f (generic function with 1 method)
1570
1571 julia> typeof(f(2))
1572 Int64
1573
1574 julia> @code_warntype f(2)
1575 MethodInstance for f(::Int64)
1576 from f(a) in Main at none:1
1577 Arguments
1578 #self#::Core.Const(f)
1579 a::Int64
1580 Body::UNION{FLOAT64, INT64}
1581 1 ─ %1 = (a > 1)::Bool
1582 └── goto #3 if not %1
1583 2 ─ return 1
1584 3 ─ return 1.0
1585
1586 julia> @inferred f(2)
1587 ERROR: return type Int64 does not match inferred return type Union{Float64, Int64}
1588 [...]
1589
1590 julia> @inferred max(1, 2)
1591 2
1592

1594 g (generic function with 1 method)
1595
1596 julia> @inferred g(20)
1597 ERROR: return type Float64 does not match inferred return type Union{Missing, Float64}
1598 [...]
1599
1600 julia> @inferred Missing g(20)
1601 1.0
1602
1603 julia> h(a) = a < 10 ? missing : f(a)
1604 h (generic function with 1 method)
1605
1606 julia> @inferred Missing h(20)
1607 ERROR: return type Int64 does not match inferred return type Union{Missing, Float64, Int64}
1608 [...]
1609 ```

1612 _inferred(ex, __module__)
1613 end
1614 macro inferred(allow, ex)
1615 _inferred(ex, __module__, allow)
1616 end
1617 function _inferred(ex, mod, allow = :(Union{}))
1618 if Meta.isexpr(ex, :ref)
1619 ex = Expr(:call, :getindex, ex.args...)
1620 end
1621 Meta.isexpr(ex, :call)|| error("@inferred requires a call expression")
1622 farg = ex.args[1]
1623 if isa(farg, Symbol) && farg !== :.. && first(string(farg)) == '.'
1624 farg = Symbol(string(farg)[2:end])
1625 ex = Expr(:call, GlobalRef(Test, :_materialize_broadcasted),
1626 farg, ex.args[2:end]...)
1627 end
1628 Base.remove_linenums!(let ex = ex;
1629 quote
1630 let
1631 allow = $(esc(allow))
1632 allow isa Type || throw(ArgumentError("@inferred requires a type as second argument"))
1633 $(if any(a->(Meta.isexpr(a, :kw) || Meta.isexpr(a, :parameters)), ex.args)
1634 # Has keywords
1635 args = gensym()
1636 kwargs = gensym()
1637 quote
1638 $(esc(args)), $(esc(kwargs)), result = $(esc(Expr(:call, _args_and_call, ex.args[2:end]..., ex.args[1])))
1639 inftypes = $(gen_call_with_extracted_types(mod, Base.return_types, :($(ex.args[1])($(args)...; $(kwargs)...))))
1640 end
1641 else
1642 # No keywords
1643 quote
1644 args = ($([esc(ex.args[i]) for i = 2:length(ex.args)]...),)
1645 result = $(esc(ex.args[1]))(args...)
1646 inftypes = Base.return_types($(esc(ex.args[1])), Base.typesof(args...))
1647 end
1648 end)
1649 @assert length(inftypes) == 1
1650 rettype = result isa Type ? Type{result} : typeof(result)
1651 rettype <: allow || rettype == typesplit(inftypes[1], allow) || error("return type $rettype does not match inferred return type $(inftypes[1])")
1652 result
1653 end
1654 end
1655 end)

1818 # find if var will be constrained to have a definite value
1819 # in any concrete leaftype subtype of typ
1820 function constrains_param(var::TypeVar, @nospecialize(typ), covariant::Bool)
1821 typ === var && return true
1822 while typ isa UnionAll
1823 covariant && constrains_param(var, typ.var.ub, covariant) && return true
1824 # typ.var.lb doesn't constrain var
1825 typ = typ.body
1826 end
1827 if typ isa Union
1828 # for unions, verify that both options would constrain var
1829 ba = constrains_param(var, typ.a, covariant)
1830 bb = constrains_param(var, typ.b, covariant)
1831 (ba && bb) && return true
1832 elseif typ isa DataType
1833 # return true if any param constrains var
1834 fc = length(typ.parameters)
1835 if fc > 0
1836 if typ.name === Tuple.name
1837 # vararg tuple needs special handling
1838 for i in 1:(fc - 1)
1839 p = typ.parameters[i]
1840 constrains_param(var, p, covariant) && return true
1841 end
1842 lastp = typ.parameters[fc]
1843 vararg = Base.unwrap_unionall(lastp)
1844 if vararg isa Core.TypeofVararg && isdefined(vararg, :N)
1845 constrains_param(var, vararg.N, covariant) && return true
1846 # T = vararg.parameters[1] doesn't constrain var
1847 else
1848 constrains_param(var, lastp, covariant) && return true
1849 end
1850 else
1851 for i in 1:fc
1852 p = typ.parameters[i]
1853 constrains_param(var, p, false) && return true
1854 end
1855 end
1856 end
1857 end
1858 return false
1859 end

1863 var = sig.var
1864 sig = sig.body
1865 if !constrains_param(var, sig, true)
1866 return true
1867 end
1868 end
1869 return false

1971 n = length(B)
1972 if N ≠ 1
