40 """
41 LazySet{N}
42
43 Abstract type for convex sets, i.e., sets characterized by a (possibly infinite)
44 intersection of halfspaces, or equivalently, sets ``S`` such that for any two
45 elements ``x, y ∈ S`` and ``0 ≤ λ ≤ 1`` it holds that ``λ·x + (1-λ)·y ∈ S``.
46
47 ### Notes
48
49 `LazySet` types should be parameterized with a type `N`, typically `N<:Real`,
50 for using different numeric types.
51
52 Every concrete `LazySet` must define the following functions:
53 - `σ(d::AbstractVector, S::LazySet)` -- the support vector of `S` in a given
54 direction `d`
55 - `dim(S::LazySet)` -- the ambient dimension of `S`
56
57 The function
58 - `ρ(d::AbstractVector, S::LazySet)` -- the support function of `S` in a given
59 direction `d`
60 is optional because there is a fallback implementation relying on `σ`.
61 However, for unbounded sets (which includes most lazy set types) this fallback
62 cannot be used and an explicit method must be implemented.
63
64 The subtypes of `LazySet` (including abstract interfaces):

314 """
315 _isbounded_unit_dimensions(S::LazySet{N}) where {N}
316
317 Determine whether a set is bounded in each unit dimension.
318
319 ### Input
320
321 - `S` -- set
322
323 ### Output
324
325 `true` iff the set is bounded in each unit dimension.
326
327 ### Algorithm
328
329 This function performs ``2n`` support function checks, where ``n`` is the
330 ambient dimension of `S`.
331 """
332 function _isbounded_unit_dimensions(S::LazySet{N}) where {N}
333 n = dim(S)
334 @inbounds for i in 1:n
335 for o in [one(N), -one(N)]
336 d = SingleEntryVector(i, n, o)
337 if ρ(d, S) == N(Inf)
338 return false
339 end
340 end
341 end
342 return true
343 end

503 """
504 ==(X::LazySet, Y::LazySet)
505
506 Return whether two LazySets of the same type are exactly equal.
507
508 ### Input
509
510 - `X` -- any `LazySet`
511 - `Y` -- another `LazySet` of the same type as `X`
512
513 ### Output
514
515 - `true` iff `X` is equal to `Y`.
516
517 ### Notes
518
519 The check is purely syntactic and the sets need to have the same base type.
520 For instance, `X::VPolytope == Y::HPolytope` returns `false` even if `X` and `Y`
521 represent the same polytope.
522 However `X::HPolytope{Int64} == Y::HPolytope{Float64}` is a valid comparison.
523
524 ### Algorithm
525
526 We recursively compare the fields of `X` and `Y` until a mismatch is found.
527
528 ### Examples
529
530 ```jldoctest
531 julia> HalfSpace([1], 1) == HalfSpace([1], 1)
532 true
533
534 julia> HalfSpace([1], 1) == HalfSpace([1.0], 1.0)
535 true
536
537 julia> Ball1([0.], 1.) == Ball2([0.], 1.)
538 false
539 ```
540 """
541 function ==(X::LazySet, Y::LazySet)
542 # if the common supertype of X and Y is abstract, they cannot be compared
543 if isabstracttype(promote_type(typeof(X), typeof(Y)))
544 return false
545 end
546
547 for f in fieldnames(typeof(X))
548 if getfield(X, f) != getfield(Y, f)
549 return false
550 end
551 end
552
553 return true
554 end

663 """
664 reflect(P::LazySet)
665
666 Concrete reflection of a convex set `P`, resulting in the reflected set `-P`.
667
668 ### Note
669
670 This function requires that the list of constraints of the set `P` is
671 available, i.e. such that it can be written as
672 ``P = \\{z ∈ ℝⁿ: ⋂ sᵢᵀz ≤ rᵢ, i = 1, ..., N\\}.``
673
674 This function can be used to implement the alternative definition of the
675 Minkowski Difference, which writes as
676 ```math
677 A ⊖ B = \\{a − b | a ∈ A, b ∈ B\\} = A ⊕ (-B)
678 ```
679 by calling `minkowski_sum(A, reflect(B))`.
680 """
681 function reflect(P::LazySet)
682 @assert applicable(constraints_list, P) "this function " *
683 "requires that the list of constraints is available, but it is not; " *
684 "if the set is bounded, try overapproximating with an `HPolytope` first"
685
686 F, g = tosimplehrep(P)
687 T = isbounded(P) ? HPolytope : HPolyhedron
688 return T(-F, g)
689 end

758 """
759 plot_recipe(X::LazySet{N}, [ε]=N(PLOT_PRECISION)) where {N}
760
761 Convert a convex set to a pair `(x, y)` of points for plotting.
762
763 ### Input
764
765 - `X` -- convex set
766 - `ε` -- (optional, default: `PLOT_PRECISION`) approximation error bound
767
768 ### Output
769
770 A pair `(x, y)` of points that can be plotted.
771
772 ### Algorithm
773
774 We first assert that `X` is bounded.
775
776 One-dimensional sets are converted to an `Interval`.
777 We do not support three-dimensional or higher-dimensional sets at the moment.
778
779 For two-dimensional sets, we first compute a polygonal overapproximation.
780 The second argument, `ε`, corresponds to the error in Hausdorff distance between
781 the overapproximating set and `X`.
782 The default value `PLOT_PRECISION` is chosen such that the unit ball in the
783 2-norm is approximated with reasonable accuracy.
784 On the other hand, if you only want to produce a fast box-overapproximation of
785 `X`, pass `ε=Inf`.
786 Finally, we use the plot recipe for polygons.
787 """
788 function plot_recipe(X::LazySet{N}, ε=N(PLOT_PRECISION)) where {N}
789 @assert dim(X) <= 2 "cannot plot a $(dim(X))-dimensional $(typeof(X))"
790 @assert isbounded(X) "cannot plot an unbounded $(typeof(X))"
791
792 if dim(X) == 1
793 Y = convert(Interval, X)
794 else
795 Y = overapproximate(X, ε)
796 end
797 return plot_recipe(Y, ε)
798 end

892 First we check `X ≈ Y`, which returns `true` if and only if `X` and `Y` have the
893 same type and approximately the same values (checked with `LazySets._isapprox`).
894 If that fails, we check the double inclusion `X ⊆ Y && Y ⊆ X`.
895
896 ### Examples
897
898 ```jldoctest
899 julia> X = BallInf([0.1, 0.2], 0.3);
900
901 julia> Y = convert(HPolytope, X);
902
903 julia> X == Y
904 false
905
906 julia> isequivalent(X, Y)
907 true
908 ```
909 """
910 function isequivalent(X::LazySet, Y::LazySet)
911 try # TODO temporary try-catch construct until ≈ is fixed for all set types
912 if X ≈ Y
913 return true
914 end
915 catch e
916 end
917 return _isequivalent_inclusion(X, Y)
918 end
919
920 function _isequivalent_inclusion(X::LazySet, Y::LazySet)
921 return X ⊆ Y && Y ⊆ X
922 end

996 """
997 area(X::LazySet{N}) where {N}
998
999 Compute the area of a two-dimensional polytopic set using the Shoelace formula.
1000
1001 ### Input
1002
1003 - `X` -- two-dimensional set
1004
1005 ### Output
1006
1007 A number representing the area of `X`.
1008
1009 ### Notes
1010
1011 This algorithm is applicable to any lazy set `X` such that its list of vertices,
1012 `vertices_list`, can be computed.
1013
1014 ### Algorithm
1015
1016 Let `m` be the number of vertices of `X`. The following instances are considered:
1017
1018 - `m = 0, 1, 2`: the output is zero.
1019 - `m = 3`: the triangle case is computed using the Shoelace formula with 3 points.
1020 - `m = 4`: the quadrilateral case is obtained by the factored version of the Shoelace
1021 formula with 4 points.
1022
1023 Otherwise, the general Shoelace formula is used; for detals see the wikipedia
1024 article [Shoelace formula](https://en.wikipedia.org/wiki/Shoelace_formula).
1025 """
1026 function area(X::LazySet{N}) where {N}
1027 @assert dim(X) == 2 "this function only applies to two-dimensional sets, " *
1028 "but the given set is $(dim(X))-dimensional"
1029
1030 Xpoly = convert(VPolygon, X)
1031 vlist = vertices_list(Xpoly)
1032 m = length(vlist)
1033
1034 if m <= 2
1035 return zero(N)
1036 end
1037
1038 if m == 3 # triangle
1039 res = _area_triangle(vlist)
1040
1041 elseif m == 4 # quadrilateral
1042 res = _area_quadrilateral(vlist)
1043
1044 else # general case
1045 res = _area_polygon(vlist)
1046 end
1047
1048 return res
1049 end
1050
1051 function _area_triangle(v::Vector{VN}) where {N, VN<:AbstractVector{N}}
1052 A = v[1]
1053 B = v[2]
1054 C = v[3]
1055 res = A[1] * (B[2] - C[2]) + B[1] * (C[2] - A[2]) + C[1] * (A[2] - B[2])
1056 return abs(res/2)
1057 end
1058
1059 function _area_quadrilateral(v::Vector{VN}) where {N, VN<:AbstractVector{N}}
1060 A = v[1]
1061 B = v[2]
1062 C = v[3]
1063 D = v[4]
1064 res = A[1] * (B[2] - D[2]) + B[1] * (C[2] - A[2]) + C[1] * (D[2] - B[2]) + D[1] * (A[2] - C[2])
1065 return abs(res/2)
1066 end
1067
1068 function _area_polygon(v::Vector{VN}) where {N, VN<:AbstractVector{N}}
1069 m = length(v)
1070 @inbounds res = v[m][1] * v[1][2] - v[1][1] * v[m][2]
1071 for i in 1:m-1
1072 @inbounds res += v[i][1] * v[i+1][2] - v[i+1][1] * v[i][2]
1073 end
1074 return abs(res/2)
1075 end

1174 delaunay(X::LazySet)
1175
1176 Compute the Delaunay triangulation of the given convex set.
1177
1178 ### Input
1179
1180 - `X` -- set
1181
1182 ### Output
1183
1184 A union of polytopes in vertex representation.
1185
1186 ### Notes
1187
1188 This function requires that you have properly installed the package
1189 [MiniQhull.jl](https://github.com/gridap/MiniQhull.jl), including the library
1190 [Qhull](http://www.qhull.org/).
1191
1192 The method works in arbitrary dimension and the requirement is that the list of
1193 vertices of `X` can be obtained.
1194 """
1195 function delaunay(X::LazySet)
1196 n = dim(X)
1197 v = vertices_list(X)
1198 m = length(v)
1199 coordinates = vcat(v...)
1200 connectivity_matrix = delaunay(n, m, coordinates)
1201 nelements = size(connectivity_matrix, 2)
1202 elements = [VPolytope(v[connectivity_matrix[:, i]]) for i in 1:nelements]
1203 return UnionSetArray(elements)
1204 end
